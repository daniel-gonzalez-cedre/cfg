        \begin{algorithm}[caption={First Four Insertion Sort}, label={alg4}]
            input:  real list $(r_1, r_2, r_3, r_4, \dots r_n)$ with $n \geq 2$
            output: real list
            begin
                for $j$ in $(2, \dots \min\set*{4, n})$:
                    $i \gets 1$
                    while $r_i < r_j$:
                        $i \gets i + 1$
                    temp $\gets r_j$
                    for $k$ in $(j, \dots i)$:
                        $r_k \gets r_{k - 1}$
                    $r_i \gets$ temp
                return $(r_1, r_2, r_3, r_4, \dots r_n)$
            end
        \end{algorithm}
\end{enumerate}
